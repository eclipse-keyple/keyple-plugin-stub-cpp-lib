137void StubReaderAdapter::insertCard(std::shared_ptr<StubSmartCard> smartCard)
138{
139 Assert::getInstance().notNull(smartCard, "smart card");
140
141 if (checkCardPresence()) {
142 mLogger->warn("You must remove the inserted card before inserted another one\n");
143 return;
144 }
145
146 const std::string protocol = smartCard->getCardProtocol();
147 if (!Arrays::contains(mActivatedProtocols, protocol)) {
148 mLogger->trace("Inserted card protocol % does not match any activated protocol, please " \
149 "use activateProtocol() method\n",
150 protocol);
151
152 return;
153 }
154
155 mLogger->trace("Inserted card %\n", smartCard);
156
157 mSmartCard = smartCard;
158}
