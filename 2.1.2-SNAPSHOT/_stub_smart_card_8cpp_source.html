103const std::vector<uint8_t> StubSmartCard::processApdu(const std::vector<uint8_t>& apduIn)
104{
105 if (apduIn.size() == 0) {
106 return std::vector<uint8_t>();
107 }
108
109 /* Convert apduIn to hex */
110 const std::string hexApdu = HexUtil::toHex(apduIn);
111
112 if (mApduResponseProvider != nullptr) {
113 const std::string responseFromRequest =
114 mApduResponseProvider->getResponseFromRequest(hexApdu);
115 if (responseFromRequest != "") {
116 return HexUtil::toByteArray(responseFromRequest);
117 }
118
119 } else if (!mHexCommands.empty()) {
120 /* Return matching hex response if the provided APDU matches the regex */
121 for (const auto& hexCommand : mHexCommands) {
122 std::unique_ptr<Pattern> p = Pattern::compile(hexCommand.first);
123 if (p->matcher(hexApdu)->matches()) {
124 return HexUtil::toByteArray(hexCommand.second);
125 }
126 }
127 }
128
129 /* Throw a CardIOException if not found */
130 throw CardIOException("No response available for this request: " + hexApdu);
131}
